%! Creates a list with sorted items
\newcommand{\sortitem}[2]{
    \DTLnewrow{list}
    \DTLnewdbentry{list}{label}{#1}
    \DTLnewdbentry{list}{description}{#2}
}

\newenvironment{sortedList}
{
    \DTLifdbexists{list}{\DTLcleardb{list}}{\DTLnewdb{list}}
    }{
    \DTLsort{label}{list}
    \begin{itemize}
        \DTLforeach*{list}{\theLabel=label,\theDesc=description}{
            \item \textbf{\theLabel} \quad \theDesc}
    \end{itemize}
}

%! Text manipulation commands

%! Bold and italic text
\newcommand{\bitext}[1]{\textbf{\textit{#1}}}

%! Appendix
\newcommand{\addappendix}{   % Self-created command to insert appendix with predefined settings
    \newpage
    \appendix
    \section*{Appendix}   % Name of appendix
    \addcontentsline{toc}{section}{Appendix}  % Add appendix name to table of contents
    \renewcommand{\thesubsection}{\Alph{subsection}}    % Change numbering of section to upper-case letters.
}

%! Front matter is located before the table of contents, e.g. preface and often has separate settings

}
